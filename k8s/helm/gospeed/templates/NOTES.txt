gospeed server has been deployed!

To get the server address:

{{- if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc {{ include "gospeed.fullname" . }}-tcp -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "gospeed -s $SERVICE_IP:{{ .Values.server.port }}"
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc {{ include "gospeed.fullname" . }}-tcp -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "gospeed -s $NODE_IP:$NODE_PORT"
{{- else }}
  kubectl port-forward svc/{{ include "gospeed.fullname" . }}-tcp {{ .Values.server.port }}:{{ .Values.server.port }} -n {{ .Release.Namespace }}
  echo "gospeed -s localhost:{{ .Values.server.port }}"
{{- end }}
